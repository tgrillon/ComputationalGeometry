10Mesh::VerticesAroundVertexCirculator::VerticesAroundVertexCirculator(const Mesh& mesh, const VertexIndex index)
11 : m_Mesh(mesh)
12 , m_CentralVertexIdx(index)
13{
14 // Get the vertex data of the central vertex.
15 const Vertex& curVertex = m_Mesh.GetVertexData(m_CentralVertexIdx);
16 assert(curVertex.IncidentTriangleIdx != -1); // Its incident triangle index must be set.
17
18 // Set the current triangle index.
19 m_CurTriangleIdx = curVertex.IncidentTriangleIdx;
20
21 // Get the current triangle data.
22 const Triangle& curFace = m_Mesh.GetTriangleData(m_CurTriangleIdx);
23 // Get the local index of the central mesh in the current triangle.
24 int localIdx = GetVertexLocalIndex(curFace, m_CentralVertexIdx);
25 assert(localIdx != -1); // The current triangle must contains the central vertex.
26
27 // Set the current vertex index.
28 m_CurVertexLocalIdx = IndexHelpers::Previous[localIdx];
29 m_CurVertexIdx = curFace.Vertices[m_CurVertexLocalIdx];
30}

32bool Mesh::VerticesAroundVertexCirculator::operator==(const Mesh::VerticesAroundVertexCirculator& rhs) const
33{
34 // Exit for counter clock-wise travel (case where the vertex neighborhood form a ring)
35 bool hasCompletedFullCirculation = m_IsInCCWOrder && m_IsActive && m_CurVertexIdx == rhs.m_CurVertexIdx;
36 // Exit for clock-wise travel (case where the vertex neighborhood is open and has a boundary)
37 bool hasReachedBoundaryEnd = !m_IsInCCWOrder && m_CurTriangleIdx == -1;
38 return hasCompletedFullCirculation || hasReachedBoundaryEnd;

46Mesh::VerticesAroundVertexCirculator& Mesh::VerticesAroundVertexCirculator::operator++()
47{
48 m_IsActive = true;
49
50 if(m_IsInCCWOrder)
51 {
52 int neighborFaceIdx;
53 {
54 const Triangle& curFace = m_Mesh.GetTriangleData(m_CurTriangleIdx);
55 neighborFaceIdx = curFace.Neighbors[IndexHelpers::Previous[m_CurVertexLocalIdx]];
56 }
57 if(neighborFaceIdx == -1)
58 { // If there is no next triangle, we reached a boundary.
59 while(m_JumpCount > 0)
60 { // We need to go back to the previous triangle and start going in the opposite direction.
61 const Triangle& curFace = m_Mesh.GetTriangleData(m_CurTriangleIdx);
62 int localIdx = GetVertexLocalIndex(curFace, m_CentralVertexIdx);
63 assert(localIdx != -1);
64 m_PrevTriangleIdx = m_CurTriangleIdx;
65 m_CurTriangleIdx = curFace.Neighbors[IndexHelpers::Previous[localIdx]];
66 m_JumpCount--;
67 }
68
69 // We are now going in the clock-wise direction.
70 m_IsInCCWOrder = false;
71 UpdateCurVertexIndexInCWOrder();
72 }
73 else // We are still going in the counter clock-wise direction.
74 {
75 // Update the jump count.
76 m_JumpCount++;
77 m_PrevTriangleIdx = m_CurTriangleIdx;
78 m_CurTriangleIdx = neighborFaceIdx;
79
80 UpdateCurVertexIndexInCCWOrder();
81 }
82 }
83 else // We are going in the clock-wise direction.
84 {
85 m_PrevTriangleIdx = m_CurTriangleIdx;
86 const Triangle& prevFace = m_Mesh.GetTriangleData(m_CurTriangleIdx);
87 m_CurTriangleIdx = prevFace.Neighbors[IndexHelpers::Next[m_CurVertexLocalIdx]];
88
89 if(m_CurTriangleIdx == -1)
90 return *this; // We reached a boundary, we stop here.
91
92 UpdateCurVertexIndexInCWOrder();
93 }
94
95 return *this;
96}
