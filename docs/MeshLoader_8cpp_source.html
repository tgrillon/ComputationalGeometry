18// Helper lambda to skip comments and empty lines
19void SkipCommentsAndWhitespace(std::ifstream& file)
20{
21 while(file.peek() == '#' || file.peek() == '\n' || file.peek() == '\r')
22 {
23 if(file.peek() == '#')
24 {
25 // Skip the entire comment line
26 file.ignore(std::numeric_limits<std::streamsize>::max(), '\n');
27 }
28 else
29 {
30 // Skip whitespace
31 file.get();
32 }
33 }
34}

39std::unique_ptr<Data::Surface::Mesh> MeshLoader::LoadOFF(const std::filesystem::path& filepath)
40{
41 std::ifstream file(filepath);
42
43 // Checking file opening
44 if(!file.is_open())
45 {
46 Error("Failed to open file: {}", filepath.string());
47 return nullptr;
48 }
49
50 // Checking file type
51 SkipCommentsAndWhitespace(file);
52 std::string type;
53 file >> type;
54 if(type != "OFF")
55 {
56 Error("Wrong file format (must be OFF) : {}", filepath.string());
57 return nullptr;
58 }
59
60 auto mesh = std::make_unique<Data::Surface::Mesh>();
61
62 // Retrieving the number of vertices / faces
63 SkipCommentsAndWhitespace(file);
64 uint32_t nVertices, nFaces, nEdges;
65 file >> nVertices >> nFaces >> nEdges;
66
67 // Reading vertices
68 uint32_t vertexIndex = 0;
69 mesh->m_Vertices.resize(nVertices);
70 for(auto&& curVertex : mesh->m_Vertices)
71 {
72 SkipCommentsAndWhitespace(file);
73 curVertex.Index = vertexIndex++;
74 file >> curVertex.Position.x >> curVertex.Position.y >> curVertex.Position.z;
75 }
76
77 // Map to store edges and their corresponding face and edge index
78 std::unordered_map<VertexPair, std::pair<IndexType, uint8_t>> neighborMap;
79
80 uint32_t faceIndex = 0;
81 mesh->m_Faces.resize(nFaces);
82 for(auto&& curFace : mesh->m_Faces)
83 {
84 SkipCommentsAndWhitespace(file);
85
86 // Create a new face
87 curFace.Index = faceIndex++;
88
89 // Set the face index
90 file >> nVertices;
91
92 for(uint8_t idx = 0; idx < nVertices; ++idx)
93 {
94 IndexType curVertexIdx;
95 file >> curVertexIdx;
96
97 // Get pointers to the vertex.
98 Vertex& curVertex = mesh->m_Vertices[curVertexIdx];
99
100 // Set IncidentFaceIdx for the vertex if it's not already the case.
101 if(curVertex.IncidentFaceIdx == -1)
102 curVertex.IncidentFaceIdx = curFace.Index;
103
104 // Set vertices of the face.
105 curFace.Vertices[idx] = curVertex.Index;
106 }
107
108 // Set neighboring faces using the edges
109 auto SetFacesNeigbhor = [&](IndexType firstVertexIdx, IndexType secondVertexIdx, uint8_t edgeIdx)
110 {
111 if(neighborMap.find({ firstVertexIdx, secondVertexIdx }) == neighborMap.end())
112 {
113 neighborMap[{ firstVertexIdx, secondVertexIdx }] = { curFace.Index, edgeIdx };
114 }
115 else // The edge with firstVertex and secondVertex is already registered in map
116 {
117 auto [faceNeighborIdx, neighborEdgeIdx] = neighborMap[{ firstVertexIdx, secondVertexIdx }];
118 mesh->m_Faces[faceNeighborIdx].Neighbors[neighborEdgeIdx] = curFace.Index;
119 curFace.Neighbors[edgeIdx] = faceNeighborIdx;
120 }
121 };
122
123 const IndexType v0Idx = curFace.Vertices[0];
124 const IndexType v1Idx = curFace.Vertices[1];
125 const IndexType v2Idx = curFace.Vertices[2];
126
127 // Edge v0-v1
128 SetFacesNeigbhor(v0Idx, v1Idx, 2);
129 // Edge v1-v2
130 SetFacesNeigbhor(v1Idx, v2Idx, 0);
131 // Edge v2-v0
132 SetFacesNeigbhor(v2Idx, v0Idx, 1);
133 }
134
135 file.close();
136 return mesh;
137}

139std::unique_ptr<Data::Surface::Mesh> MeshLoader::LoadOBJ(const std::filesystem::path& filepath)
140{
141 std::ifstream file(filepath);
142
143 // Checking file opening
144 if(!file.is_open())
145 {
146 Error("Failed to open file: {}", filepath.string());
147 return nullptr;
148 }
149
150 // Check file extension
151 if(filepath.extension() != ".obj")
152 {
153 Error("Wrong file extension (must be .obj): {}", filepath.string());
154 return nullptr;
155 }
156
157 auto mesh = std::make_unique<Data::Surface::Mesh>();
158
159 // Map to store edges and their corresponding face and edge index
160 std::unordered_map<VertexPair, std::pair<IndexType, uint8_t>> neighborMap;
161
162 std::string type;
163 IndexType curIndexV = 0;
164 IndexType curIndexVt = 0;
165 IndexType curIndexVn = 0;
166 IndexType curIndexF = 0;
167 while(file.peek() != EOF)
168 {
169 SkipCommentsAndWhitespace(file);
170
171 if(file.peek() == EOF)
172 break;
173
174 file >> type;
175 // Process line based on its type
176 if(type == "v")
177 { // Vertex position
178 Vertex curVertex;
179 curVertex.Index = curIndexV++;
180 file >> curVertex.Position.x >> curVertex.Position.y >> curVertex.Position.z;
181 mesh->m_Vertices.emplace_back(curVertex);
182 mesh->m_VertexExtraData.emplace_back();
183 }
184 else if(type == "vt")
185 { // Vertex texture coordinate
186 const VertexProxy& curVertex = mesh->GetVertex(curIndexVt++);
187 auto texCoordED = curVertex.GetOrCreateExtraData<Vec2>();
188 file >> texCoordED.x >> texCoordED.y;
189 }
190 else if(type == "vn")
191 { // Vertex normal
192 const VertexProxy& curVertex = mesh->GetVertex(curIndexVn++);
193 auto normalED = curVertex.GetOrCreateExtraData<Vec3>();
194 file >> normalED.x >> normalED.y >> normalED.z;
195 }
196 else if(type == "f")
197 { // Face (triangle)
198 Face& curFace = mesh->m_Faces.emplace_back();
199 curFace.Index = curIndexF++;
200
201 // Read three vertices for the triangle
202 for(uint8_t idx = 0; idx < 3; idx++)
203 {
204 IndexType vertexIdx;
205 char buffer[16];
206 file >> std::ws; // Skip leading whitespace
207
208 // Read until first '/' or whitespace
209 int i = 0;
210 while(file.peek() != EOF && !std::isspace(file.peek()) && file.peek() != '/')
211 {
212 buffer[i++] = file.get();
213 }
214 buffer[i] = '\0';
215
216 // Convert string to index
217 vertexIdx = std::stoi(buffer);
218
219 Vertex& curVertex = mesh->m_Vertices[vertexIdx - 1];
220 if(curVertex.IncidentFaceIdx == -1)
221 curVertex.IncidentFaceIdx = curFace.Index;
222
223 curFace.Vertices[idx] = curVertex.Index;
224
225 // Skip texture/normal indices until next vertex or whitespace
226 while(file.peek() != EOF && !std::isspace(file.peek()))
227 {
228 file.ignore(1);
229 }
230 }
231
232 // Set neighboring faces using the edges
233 auto SetFacesNeigbhor = [&](IndexType firstVertexIdx, IndexType secondVertexIdx, uint8_t edgeIdx)
234 {
235 if(neighborMap.find({ firstVertexIdx, secondVertexIdx }) == neighborMap.end())
236 {
237 neighborMap[{ firstVertexIdx, secondVertexIdx }] = { curFace.Index, edgeIdx };
238 }
239 else // The edge with firstVertex and secondVertex is already registered in map
240 {
241 auto [faceNeighborIdx, neighborEdgeIdx] = neighborMap[{ firstVertexIdx, secondVertexIdx }];
242 mesh->m_Faces[faceNeighborIdx].Neighbors[neighborEdgeIdx] = curFace.Index;
243 curFace.Neighbors[edgeIdx] = faceNeighborIdx;
244 }
245 };
246
247 const IndexType v0Idx = curFace.Vertices[0];
248 const IndexType v1Idx = curFace.Vertices[1];
249 const IndexType v2Idx = curFace.Vertices[2];
250
251 // Edge v0-v1
252 SetFacesNeigbhor(v0Idx, v1Idx, 2);
253 // Edge v1-v2
254 SetFacesNeigbhor(v1Idx, v2Idx, 0);
255 // Edge v2-v0
256 SetFacesNeigbhor(v2Idx, v0Idx, 1);
257 }
258 }
259
260 file.close();
261 return mesh;
262}
