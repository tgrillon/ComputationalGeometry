99void Mesh::UpdateMeshConnectivity()
100{
101 std::unordered_map<VertexPair, std::pair<FaceIndex, EdgeIndex>> neighborMap;
102
103 for(FaceIndex iFace = 0; iFace < GetFaceCount(); ++iFace)
104 {
105 Face& curFace = m_Faces[iFace];
106
107 // Set the incident face index for each vertex if it's not already the case.
108 for(EdgeIndex iEdge = 0; iEdge < 3; ++iEdge)
109 {
110 int curVertexIdx = curFace.Vertices[iEdge];
111 assert(curVertexIdx != -1);
112
113 Vertex& curVertex = m_Vertices[curVertexIdx];
114
115 if(curVertex.IncidentFaceIdx == -1)
116 curVertex.IncidentFaceIdx = iFace;
117 }
118
119 // Set neighboring faces using the edges
120 auto SetFacesNeigbhor = [&](VertexIndex firstVertexIdx, VertexIndex secondVertexIdx, uint8_t edgeIdx)
121 {
122 if(neighborMap.find({ firstVertexIdx, secondVertexIdx }) == neighborMap.end())
123 {
124 neighborMap[{ firstVertexIdx, secondVertexIdx }] = { iFace, edgeIdx };
125 }
126 else // The edge with firstVertex and secondVertex is already registered in map
127 {
128 auto [faceNeighborIdx, neighborEdgeIdx] = neighborMap[{ firstVertexIdx, secondVertexIdx }];
129 m_Faces[faceNeighborIdx].Neighbors[neighborEdgeIdx] = iFace;
130 curFace.Neighbors[edgeIdx] = faceNeighborIdx;
131 }
132 };
133
134 const VertexIndex v0Idx = static_cast<VertexIndex>(curFace.Vertices[0]);
135 const VertexIndex v1Idx = static_cast<VertexIndex>(curFace.Vertices[1]);
136 const VertexIndex v2Idx = static_cast<VertexIndex>(curFace.Vertices[2]);
137
138 // Edge v0-v1
139 SetFacesNeigbhor(v0Idx, v1Idx, 2);
140 // Edge v1-v2
141 SetFacesNeigbhor(v1Idx, v2Idx, 0);
142 // Edge v2-v0
143 SetFacesNeigbhor(v2Idx, v0Idx, 1);
144 }
145}

178Mesh::VerticesAroundVertexCirculator::VerticesAroundVertexCirculator(const Mesh& mesh, const VertexIndex index)
179 : m_Mesh(mesh)
180 , m_CentralVertexIdx(index)
181{
182 // Get the vertex data of the central vertex.
183 const Vertex& curVertex = m_Mesh.GetVertexData(m_CentralVertexIdx);
184 assert(curVertex.IncidentFaceIdx != -1); // Its incident face index must be set.
185
186 // Set the current face index.
187 m_CurFaceIdx = curVertex.IncidentFaceIdx;
188
189 // Get the current face data.
190 const Face& curFace = m_Mesh.GetFaceData(m_CurFaceIdx);
191 // Get the local index of the central mesh in the current face.
192 int localIdx = GetVertexLocalIndex(curFace, m_CentralVertexIdx);
193 assert(localIdx != -1); // The current face must contains the central vertex.
194
195 // Set the current vertex index.
196 m_CurVertexLocalIdx = IndexHelpers::Previous[localIdx];
197 m_CurVertexIdx = curFace.Vertices[m_CurVertexLocalIdx];
198}

200bool Mesh::VerticesAroundVertexCirculator::operator==(const Mesh::VerticesAroundVertexCirculator& rhs) const
201{
202 // Exit for counter clock-wise travel (case where the vertex neighborhood form a ring)
203 bool hasCompletedFullCirculation = m_IsInCCWOrder && m_IsActive && m_CurVertexIdx == rhs.m_CurVertexIdx;
204 // Exit for clock-wise travel (case where the vertex neighborhood is open and has a boundary)
205 bool hasReachedBoundaryEnd = !m_IsInCCWOrder && m_CurFaceIdx == -1;
206 return hasCompletedFullCirculation || hasReachedBoundaryEnd;

214Mesh::VerticesAroundVertexCirculator& Mesh::VerticesAroundVertexCirculator::operator++()
215{
216 m_IsActive = true;
217
218 if(m_IsInCCWOrder)
219 {
220 int neighborFaceIdx;
221 {
222 const Face& curFace = m_Mesh.GetFaceData(m_CurFaceIdx);
223 neighborFaceIdx = curFace.Neighbors[IndexHelpers::Previous[m_CurVertexLocalIdx]];
224 }
225 if(neighborFaceIdx == -1)
226 { // If there is no next face, we reached a boundary.
227 while(m_JumpCount > 0)
228 { // We need to go back to the previous face and start going in the opposite direction.
229 const Face& curFace = m_Mesh.GetFaceData(m_CurFaceIdx);
230 int localIdx = GetVertexLocalIndex(curFace, m_CentralVertexIdx);
231 assert(localIdx != -1);
232 m_PrevFaceIdx = m_CurFaceIdx;
233 m_CurFaceIdx = curFace.Neighbors[IndexHelpers::Previous[localIdx]];
234 m_JumpCount--;
235 }
236
237 // We are now going in the clock-wise direction.
238 m_IsInCCWOrder = false;
239 UpdateCurVertexIndexInCWOrder();
240 }
241 else // We are still going in the counter clock-wise direction.
242 {
243 // Update the jump count.
244 m_JumpCount++;
245 m_PrevFaceIdx = m_CurFaceIdx;
246 m_CurFaceIdx = neighborFaceIdx;
247
248 UpdateCurVertexIndexInCCWOrder();
249 }
250 }
251 else // We are going in the clock-wise direction.
252 {
253 m_PrevFaceIdx = m_CurFaceIdx;
254 const Face& prevFace = m_Mesh.GetFaceData(m_CurFaceIdx);
255 m_CurFaceIdx = prevFace.Neighbors[IndexHelpers::Next[m_CurVertexLocalIdx]];
256
257 if(m_CurFaceIdx == -1)
258 return *this; // We reached a boundary, we stop here.
259
260 UpdateCurVertexIndexInCWOrder();
261 }
262
263 return *this;
264}
