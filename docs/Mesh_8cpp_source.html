109void Mesh::UpdateMeshConnectivity()
110{
111 std::unordered_map<VertexPair, std::pair<TriangleIndex, EdgeIndex>> neighborMap;
112
113 for(TriangleIndex iTriangle = 0; iTriangle < GetTriangleCount(); ++iTriangle)
114 {
115 Triangle& curTriangle = m_Triangles[iTriangle];
116
117 // Set the incident triangle index for each vertex if it's not already the case.
118 for(EdgeIndex iEdge = 0; iEdge < 3; ++iEdge)
119 {
120 int curVertexIdx = curTriangle.Vertices[iEdge];
121 assert(curVertexIdx != -1);
122
123 Vertex& curVertex = m_Vertices[curVertexIdx];
124
125 if(curVertex.IncidentTriangleIdx == -1)
126 curVertex.IncidentTriangleIdx = iTriangle;
127 }
128
129 // Set neighboring faces using the edges
130 auto SetFacesNeigbhor = [&](VertexIndex firstVertexIdx, VertexIndex secondVertexIdx, uint8_t edgeIdx)
131 {
132 if(neighborMap.find({ firstVertexIdx, secondVertexIdx }) == neighborMap.end())
133 {
134 neighborMap[{ firstVertexIdx, secondVertexIdx }] = { iTriangle, edgeIdx };
135 }
136 else // The edge with firstVertex and secondVertex is already registered in map
137 {
138 auto [faceNeighborIdx, neighborEdgeIdx] = neighborMap[{ firstVertexIdx, secondVertexIdx }];
139 m_Triangles[faceNeighborIdx].Neighbors[neighborEdgeIdx] = iTriangle;
140 curTriangle.Neighbors[edgeIdx] = faceNeighborIdx;
141 }
142 };
143
144 const VertexIndex v0Idx = static_cast<VertexIndex>(curTriangle.Vertices[0]);
145 const VertexIndex v1Idx = static_cast<VertexIndex>(curTriangle.Vertices[1]);
146 const VertexIndex v2Idx = static_cast<VertexIndex>(curTriangle.Vertices[2]);
147
148 // Edge v0-v1
149 SetFacesNeigbhor(v0Idx, v1Idx, 2);
150 // Edge v1-v2
151 SetFacesNeigbhor(v1Idx, v2Idx, 0);
152 // Edge v2-v0
153 SetFacesNeigbhor(v2Idx, v0Idx, 1);
154 }

187void Mesh::ComputeTriangleNormals(bool normalize)
188{
189 // Add extra data containers for triangles if necessary.
190 if(!HasTrianglesExtraDataContainer())
191 AddTrianglesExtraDataContainer();
192
193 for(TriangleIndex iTriangle = 0; iTriangle < GetTriangleCount(); ++iTriangle)
194 {
195 const TriangleProxy& curTriangle = GetTriangle(iTriangle);
196
197 // Get each vertex position.
198 const Vec3& posA = m_Vertices[curTriangle.GetVertex(0)].Position;
199 const Vec3& posB = m_Vertices[curTriangle.GetVertex(1)].Position;
200 const Vec3& posC = m_Vertices[curTriangle.GetVertex(2)].Position;
201
202 const Vec3 AB = Normalize(posB - posA);
203 const Vec3 AC = Normalize(posC - posA);
204
205 // Compute and store the normal as an extra data to the current triangle.
206 TriangleNormalExtraData& curTriangleNormal = curTriangle.GetOrCreateExtraData<TriangleNormalExtraData>();
207 Vec3 computedNormal = Cross(AB, AC);
208 if(normalize)
209 computedNormal = Normalize(computedNormal);
210 curTriangleNormal.SetData(computedNormal);
211 }
212}

214void Mesh::ComputeSmoothVertexNormals(bool normalize)
215{
216 // Add extra data containers for triangles if necessary.
217 if(!HasTrianglesExtraDataContainer())
218 AddTrianglesExtraDataContainer();
219
220 // Add extra data containers for vertices if necessary.
221 if(!HasVerticesExtraDataContainer())
222 AddVerticesExtraDataContainer();
223
224 for(TriangleIndex iTriangle = 0; iTriangle < GetTriangleCount(); ++iTriangle)
225 {
226 const TriangleProxy& curTriangle = GetTriangle(iTriangle);
227
228 // Get each vertex position.
229 const Vec3& posA = m_Vertices[curTriangle.GetVertex(0)].Position;
230 const Vec3& posB = m_Vertices[curTriangle.GetVertex(1)].Position;
231 const Vec3& posC = m_Vertices[curTriangle.GetVertex(2)].Position;
232
233 const Vec3 AB = Normalize(posB - posA);
234 const Vec3 AC = Normalize(posC - posA);
235
236 // Get or compute the current triangle normal.
237 Vec3 curTriangleNormal;
238 if(curTriangle.HasExtraData<TriangleNormalExtraData>())
239 {
240 curTriangleNormal = curTriangle.GetExtraData<TriangleNormalExtraData>()->GetData();
241 }
242 else
243 {
244 curTriangleNormal = Cross(AB, AC);
245 }
246
247 curTriangleNormal = Normalize(curTriangleNormal);
248
249 // Compute the remaining vectors of the triangle.
250 const Vec3 BC = glm::normalize(posC - posB);
251 const Vec3 BA = glm::normalize(posA - posB);
252
253 const Vec3 CA = glm::normalize(posA - posC);
254 const Vec3 CB = glm::normalize(posB - posC);
255
256 // Compute each angle (in randian).
257 const float angleA = Angle(AB, AC);
258 const float angleB = Angle(BC, BA);
259 const float angleC = Angle(CA, CB);
260
261 // Get a proxy on each vertex of the triangle.
262 const VertexProxy& vertexA = GetVertex(curTriangle.GetVertex(0));
263 const VertexProxy& vertexB = GetVertex(curTriangle.GetVertex(1));
264 const VertexProxy& vertexC = GetVertex(curTriangle.GetVertex(2));
265
266 // Add the normal weighted by the related angle as an extra data to each vertex.
267 FlatVertexNormalsExtraData& vertexAExtraData = vertexA.GetOrCreateExtraData<FlatVertexNormalsExtraData>();
268 vertexAExtraData.GetData().emplace_back(curTriangleNormal * angleA);
269 FlatVertexNormalsExtraData& vertexBExtraData = vertexB.GetOrCreateExtraData<FlatVertexNormalsExtraData>();
270 vertexBExtraData.GetData().emplace_back(curTriangleNormal * angleB);
271 FlatVertexNormalsExtraData& vertexCExtraData = vertexC.GetOrCreateExtraData<FlatVertexNormalsExtraData>();
272 vertexCExtraData.GetData().emplace_back(curTriangleNormal * angleC);
273 }
274
275 // Compute the smooth normal for each vertex of the mesh.
276 for(VertexIndex iVertex = 0; iVertex < GetVertexCount(); ++iVertex)
277 {
278 // Get the current vertex and create the extra data that will handle the smooth vertex normal.
279 const VertexProxy& curVertex = GetVertex(iVertex);
280 SmoothVertexNormalExtraData& curVertexNormal = curVertex.GetOrCreateExtraData<SmoothVertexNormalExtraData>();
281
282 // Get the precomputed flat vertex normals weighted by the angles.
283 auto vertexFlatNormals = curVertex.GetExtraData<FlatVertexNormalsExtraData>();
284 assert(vertexFlatNormals != nullptr);
285
286 // Accumulate the weighted flat vertex normals.
287 Vec3 computedNormal = std::accumulate(
288 vertexFlatNormals->GetData().begin(),
289 vertexFlatNormals->GetData().end(),
290 Vec3{ 0., 0., 0. },
291 std::plus<Vec3>());
292
293 if(normalize)
294 computedNormal = Normalize(computedNormal);
295 curVertexNormal.SetData(computedNormal);
296
297 // Erase extra data that is used to compute smooth vertex normal.
298 curVertex.EraseExtraData<FlatVertexNormalsExtraData>();
299 }

302void Mesh::UpdateVerticesBoundaryStatus()
303{
304 // Add extra data containers for vertices if necessary.
305 if(!HasVerticesExtraDataContainer())
306 AddVerticesExtraDataContainer();
307
308 for(VertexIndex iVertex = 0; iVertex < GetVertexCount(); ++iVertex)
309 {
310 VertexProxy curVertex = GetVertex(iVertex);
311 auto& curBoundaryStatus = curVertex.GetOrCreateExtraData<IsBoundaryVertexExtraData>();
312 curBoundaryStatus.SetData(false);
313 }
314
315 for(auto&& curTriangle : m_Triangles)
316 {
317 for(EdgeIndex iEdge = 0; iEdge < 3; ++iEdge)
318 {
319 // If the edge opposite to the current vertex is a boundary edge,
320 // set the boundary status on the two incident vertices.
321 if(curTriangle.Neighbors[iEdge] == -1)
322 {
323 VertexProxy nextVertex = GetVertex(curTriangle.Vertices[IndexHelpers::Next[iEdge]]);
324 auto& nextBoundaryStatus = nextVertex.GetOrCreateExtraData<IsBoundaryVertexExtraData>();
325 nextBoundaryStatus.SetData(true);
326
327 VertexProxy previousVertex = GetVertex(curTriangle.Vertices[IndexHelpers::Previous[iEdge]]);
328 auto& previousBoundaryStatus = previousVertex.GetOrCreateExtraData<IsBoundaryVertexExtraData>();
329 previousBoundaryStatus.SetData(true);
330 }
331 }
332 }
