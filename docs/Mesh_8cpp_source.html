106void Mesh::UpdateMeshConnectivity()
107{
108 std::unordered_map<VertexPair, std::pair<TriangleIndex, EdgeIndex>> neighborMap;
109
110 for(TriangleIndex iTriangle = 0; iTriangle < GetTriangleCount(); ++iTriangle)
111 {
112 Triangle& curFace = m_Triangles[iTriangle];
113
114 // Set the incident triangle index for each vertex if it's not already the case.
115 for(EdgeIndex iEdge = 0; iEdge < 3; ++iEdge)
116 {
117 int curVertexIdx = curFace.Vertices[iEdge];
118 assert(curVertexIdx != -1);
119
120 Vertex& curVertex = m_Vertices[curVertexIdx];
121
122 if(curVertex.IncidentTriangleIdx == -1)
123 curVertex.IncidentTriangleIdx = iTriangle;
124 }
125
126 // Set neighboring faces using the edges
127 auto SetFacesNeigbhor = [&](VertexIndex firstVertexIdx, VertexIndex secondVertexIdx, uint8_t edgeIdx)
128 {
129 if(neighborMap.find({ firstVertexIdx, secondVertexIdx }) == neighborMap.end())
130 {
131 neighborMap[{ firstVertexIdx, secondVertexIdx }] = { iTriangle, edgeIdx };
132 }
133 else // The edge with firstVertex and secondVertex is already registered in map
134 {
135 auto [faceNeighborIdx, neighborEdgeIdx] = neighborMap[{ firstVertexIdx, secondVertexIdx }];
136 m_Triangles[faceNeighborIdx].Neighbors[neighborEdgeIdx] = iTriangle;
137 curFace.Neighbors[edgeIdx] = faceNeighborIdx;
138 }
139 };
140
141 const VertexIndex v0Idx = static_cast<VertexIndex>(curFace.Vertices[0]);
142 const VertexIndex v1Idx = static_cast<VertexIndex>(curFace.Vertices[1]);
143 const VertexIndex v2Idx = static_cast<VertexIndex>(curFace.Vertices[2]);
144
145 // Edge v0-v1
146 SetFacesNeigbhor(v0Idx, v1Idx, 2);
147 // Edge v1-v2
148 SetFacesNeigbhor(v1Idx, v2Idx, 0);
149 // Edge v2-v0
150 SetFacesNeigbhor(v2Idx, v0Idx, 1);
151 }
152}

184void Mesh::ComputeTriangleNormals(bool computeSmoothVertexNormals)
185{
186 if(m_TrianglesExtraDataContainer.empty())
187 AddTrianglesExtraDataContainer();
188
189 if(computeSmoothVertexNormals && m_VerticesExtraDataContainer.empty())
190 AddVerticesExtraDataContainer();
191
192 for(TriangleIndex iTriangle = 0; iTriangle < GetTriangleCount(); ++iTriangle)
193 {
194 const TriangleProxy& curFace = GetTriangle(iTriangle);
195
196 const Vec3& posA = m_Vertices[curFace.GetVertex(0)].Position;
197 const Vec3& posB = m_Vertices[curFace.GetVertex(1)].Position;
198 const Vec3& posC = m_Vertices[curFace.GetVertex(2)].Position;
199
200 const Vec3 AB = glm::normalize(posB - posA);
201 const Vec3 AC = glm::normalize(posC - posA);
202
203 TriangleNormalExtraData& curFaceNormal = curFace.GetOrCreateExtraData<TriangleNormalExtraData>();
204 const Vec3& computedNormal = glm::normalize(glm::cross(AB, AC));
205 curFaceNormal.SetData(computedNormal);
206
207 if(computeSmoothVertexNormals)
208 {
209 // Compute the remaining vectors of the triangle.
210 const Vec3 BC = glm::normalize(posC - posB);
211 const Vec3 BA = glm::normalize(posA - posB);
212
213 const Vec3 CA = glm::normalize(posA - posC);
214 const Vec3 CB = glm::normalize(posB - posC);
215
216 // Compute each angle (in randian).
217 const float angleA = glm::angle(AB, AC);
218 const float angleB = glm::angle(BC, BA);
219 const float angleC = glm::angle(CA, CB);
220
221 // Get a proxy on each vertex of the triangle.
222 const VertexProxy& vertexA = GetVertex(curFace.GetVertex(0));
223 const VertexProxy& vertexB = GetVertex(curFace.GetVertex(1));
224 const VertexProxy& vertexC = GetVertex(curFace.GetVertex(2));
225
226 // Add the normal weighted by the related angle as an extra data to each vertex.
227 FlatVertexNormalsExtraData& vertexAExtraData = vertexA.GetOrCreateExtraData<FlatVertexNormalsExtraData>();
228 vertexAExtraData.GetData().emplace_back(computedNormal * angleA);
229 FlatVertexNormalsExtraData& vertexBExtraData = vertexB.GetOrCreateExtraData<FlatVertexNormalsExtraData>();
230 vertexBExtraData.GetData().emplace_back(computedNormal * angleB);
231 FlatVertexNormalsExtraData& vertexCExtraData = vertexC.GetOrCreateExtraData<FlatVertexNormalsExtraData>();
232 vertexCExtraData.GetData().emplace_back(computedNormal * angleC);
233 }
234 }
235
236 if(computeSmoothVertexNormals)
237 {
238 // Compute the smooth normal for each vertex of the mesh.
239 for(VertexIndex iVertex = 0; iVertex < GetVertexCount(); ++iVertex)
240 {
241 // Get the current vertex and create the extra data that will handle the smooth vertex normal.
242 const VertexProxy& curVertex = GetVertex(iVertex);
243 SmoothVertexNormalExtraData& curVertexNormal = curVertex.GetOrCreateExtraData<SmoothVertexNormalExtraData>();
244
245 // Get the precomputed flat vertex normals.
246 auto vertexFlatNormals = curVertex.GetExtraData<FlatVertexNormalsExtraData>();
247 assert(vertexFlatNormals != nullptr);
248
249 // Accumulate the flat vertex normals.
250 Vec3 computedNormal = std::accumulate(
251 vertexFlatNormals->GetData().begin(),
252 vertexFlatNormals->GetData().end(),
253 Vec3{ 0., 0., 0. },
254 std::plus<Vec3>());
255
256 curVertexNormal.SetData(glm::normalize(computedNormal));
257 curVertex.EraseExtraData<FlatVertexNormalsExtraData>();
258 }
259 }
260}
