101void Mesh::UpdateMeshConnectivity()
102{
103 std::unordered_map<VertexPair, std::pair<FaceIndex, EdgeIndex>> neighborMap;
104
105 for(FaceIndex iFace = 0; iFace < GetFaceCount(); ++iFace)
106 {
107 Face& curFace = m_Faces[iFace];
108
109 // Set the incident face index for each vertex if it's not already the case.
110 for(EdgeIndex iEdge = 0; iEdge < 3; ++iEdge)
111 {
112 int curVertexIdx = curFace.Vertices[iEdge];
113 assert(curVertexIdx != -1);
114
115 Vertex& curVertex = m_Vertices[curVertexIdx];
116
117 if(curVertex.IncidentFaceIdx == -1)
118 curVertex.IncidentFaceIdx = iFace;
119 }
120
121 // Set neighboring faces using the edges
122 auto SetFacesNeigbhor = [&](VertexIndex firstVertexIdx, VertexIndex secondVertexIdx, uint8_t edgeIdx)
123 {
124 if(neighborMap.find({ firstVertexIdx, secondVertexIdx }) == neighborMap.end())
125 {
126 neighborMap[{ firstVertexIdx, secondVertexIdx }] = { iFace, edgeIdx };
127 }
128 else // The edge with firstVertex and secondVertex is already registered in map
129 {
130 auto [faceNeighborIdx, neighborEdgeIdx] = neighborMap[{ firstVertexIdx, secondVertexIdx }];
131 m_Faces[faceNeighborIdx].Neighbors[neighborEdgeIdx] = iFace;
132 curFace.Neighbors[edgeIdx] = faceNeighborIdx;
133 }
134 };
135
136 const VertexIndex v0Idx = static_cast<VertexIndex>(curFace.Vertices[0]);
137 const VertexIndex v1Idx = static_cast<VertexIndex>(curFace.Vertices[1]);
138 const VertexIndex v2Idx = static_cast<VertexIndex>(curFace.Vertices[2]);
139
140 // Edge v0-v1
141 SetFacesNeigbhor(v0Idx, v1Idx, 2);
142 // Edge v1-v2
143 SetFacesNeigbhor(v1Idx, v2Idx, 0);
144 // Edge v2-v0
145 SetFacesNeigbhor(v2Idx, v0Idx, 1);
146 }
147}

180Mesh::VerticesAroundVertexCirculator::VerticesAroundVertexCirculator(const Mesh& mesh, const VertexIndex index)
181 : m_Mesh(mesh)
182 , m_CentralVertexIdx(index)
183{
184 // Get the vertex data of the central vertex.
185 const Vertex& curVertex = m_Mesh.GetVertexData(m_CentralVertexIdx);
186 assert(curVertex.IncidentFaceIdx != -1); // Its incident face index must be set.
187
188 // Set the current face index.
189 m_CurFaceIdx = curVertex.IncidentFaceIdx;
190
191 // Get the current face data.
192 const Face& curFace = m_Mesh.GetFaceData(m_CurFaceIdx);
193 // Get the local index of the central mesh in the current face.
194 int localIdx = GetVertexLocalIndex(curFace, m_CentralVertexIdx);
195 assert(localIdx != -1); // The current face must contains the central vertex.
196
197 // Set the current vertex index.
198 m_CurVertexLocalIdx = IndexHelpers::Previous[localIdx];
199 m_CurVertexIdx = curFace.Vertices[m_CurVertexLocalIdx];
200}

202bool Mesh::VerticesAroundVertexCirculator::operator==(const Mesh::VerticesAroundVertexCirculator& rhs) const
203{
204 // Exit for counter clock-wise travel (case where the vertex neighborhood form a ring)
205 bool hasCompletedFullCirculation = m_IsInCCWOrder && m_IsActive && m_CurVertexIdx == rhs.m_CurVertexIdx;
206 // Exit for clock-wise travel (case where the vertex neighborhood is open and has a boundary)
207 bool hasReachedBoundaryEnd = !m_IsInCCWOrder && m_CurFaceIdx == -1;
208 return hasCompletedFullCirculation || hasReachedBoundaryEnd;
209}

216Mesh::VerticesAroundVertexCirculator& Mesh::VerticesAroundVertexCirculator::operator++()
217{
218 m_IsActive = true;
219
220 const Face& curFace = m_Mesh.GetFaceData(m_CurFaceIdx);
221 int neighborFaceIdx = curFace.Neighbors[IndexHelpers::Previous[m_CurVertexLocalIdx]];
222 if(m_IsInCCWOrder)
223 {
224 if(neighborFaceIdx == -1)
225 { // If there is no next face, we reached a boundary.
226 while(m_JumpCount > 0)
227 { // We need to go back to the previous face and start going in the opposite direction.
228 const Face& curFace = m_Mesh.GetFaceData(m_CurFaceIdx);
229 int localIdx = GetVertexLocalIndex(curFace, m_CentralVertexIdx);
230 assert(localIdx != -1);
231 m_PrevFaceIdx = m_CurFaceIdx;
232 m_CurFaceIdx = curFace.Neighbors[IndexHelpers::Previous[localIdx]];
233 m_JumpCount--;
234 }
235
236 // We are now going in the clock-wise direction.
237 m_IsInCCWOrder = false;
238 UpdateCurVertexIndexInCWOrder();
239 }
240 else // We are still going in the counter clock-wise direction.
241 {
242 // Update the jump count.
243 m_JumpCount++;
244 m_PrevFaceIdx = m_CurFaceIdx;
245 m_CurFaceIdx = neighborFaceIdx;
246
247 UpdateCurVertexIndexInCCWOrder();
248 }
249 }
250 else // We are going in the clock-wise direction.
251 {
252 m_PrevFaceIdx = m_CurFaceIdx;
253 const Face& prevFace = m_Mesh.GetFaceData(m_CurFaceIdx);
254 m_CurFaceIdx = prevFace.Neighbors[IndexHelpers::Next[m_CurVertexLocalIdx]];
255 if(m_CurFaceIdx == -1)
256 return *this; // We reached a boundary, we stop here.
257
258 UpdateCurVertexIndexInCWOrder();
259 }
260
261 return *this;
262}
